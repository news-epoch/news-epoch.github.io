阅读时长: 3 分钟

Links:依赖冲突问题

一、简介

1 常用注解

注解 作用
@Api(tags = “01. 相册管理模块”) 添加在控制器类上，通过此注解的tags属性可配置模块名称，可以在每个模块名称之前添加数字，则API文档会根据此名称（名称前面的数字）升序排列
@ApiOperation(“添加相册”) 添加在处理请求的方法上，通过此注解的value属性可配置业务名称
@ApiOperationSupport(order = 100) 添加在处理请求的方法上，通过此注解的order属性可配置各业务的排序，最终将按照order属性的值（int类型）升序排列
@ApiModelProperty(value = “相册名称”, required = true) 添加在POJO类型的属性上，通过此注解的value属性可配置请求参数的描述文本，通过此注解的required属性可配置是否必须提交此参数（注意：此配置并不具备检查功能），通过此注解的example属性可以配置示例值
@ApiImplicitParam(name = “id”, value = “相册ID”, required = true, dataType = “long”) 添加在处理请求的方法上，通过此注解的name属性表示你需要配置方法的哪个请求参数，通过此注解的value属性配置请求参数的描述文本，通过此注解的required属性可配置是否必须提交此参数，通过此注解的dataType属性配置请求参数的数据类型，通过此注解的example属性可以配置示例值
@ApiImplicitParams({
@ApiImplicitParam(name = “id”, value = “相册ID”, required = true, dataType = “long”), @ApiImplicitParam(name = “username”, value = “用户名”, required = true) })		 添加在处理请求的方法上，当存在多个请求参数的说明是通过@ApiImplicitParam进行配置时，应该将多个@ApiImplicitParam配置作为当前注解的参数值
@JsonInclude(JsonInclude.Include.NON_NULL) 去除响应结果中的null 当需要“去除响应时为null的数据”时，可以在对应的属性上添加@JsonInclude注解进行配置，当注解属性取值为NON_NULL时，表示“不为null时，JSON结果中将包含此属性”，

@Configuration  
@EnableSwagger2WebMvc  
@EnableKnife4j     //开启 knife4j，可以不写  
@EnableAutoConfiguration  
@ConditionalOnProperty(name = "swagger.enable", matchIfMissing = true)  
public class Knife4jConfiguration {  
    @Bean  
    public Docket docket() {  
        // 创建一个 swagger 的 bean 实例  
        return new Docket(DocumentationType.SWAGGER_2)  
                // 配置基本信息  
                .apiInfo(apiInfo())  
                // 配置接口信息，设置扫描接口  
                .select()  
                // 此包路径下的类，才生成接口文档  
                .apis(RequestHandlerSelectors.basePackage("com.changgou"))  
                // 加了 RestController 注解的类，才生成接口文档  
                .apis(RequestHandlerSelectors.withClassAnnotation(RestController.class))  
                /*  
                 * PathSelectors                 *      
                 *      .any() // 满足条件的路径，该断言总为true  
                 *      .none() // 不满足条件的路径，该断言总为false（可用于生成环境屏蔽 swagger）  
                 *      .ant("/user/**") // 满足字符串表达式路径  
                 *      .regex("") // 符合正则的路径  
                 */  
                .paths(PathSelectors.any())  
                .build()  
                ;  
    }  
  
    // api基本信息  
    private ApiInfo apiInfo() {  
        return new ApiInfoBuilder()  
                // 标题  
                .title("畅购-接口文档")  
                // 描述  
                .description("众里寻他千百度，慕然回首那人却在灯火阑珊处")  
                // 服务条款链接  
                .termsOfServiceUrl("https://www.baidu.com")  
                // 许可证  
                .license("swagger-接口文档")  
                // 许可证链接  
                .licenseUrl("https://blog.csdn.net/xhmico/article/details/125353535")  
                // 联系我  
                .contact(new Contact(  
                        "陈先生",  
                        "https://blog.csdn.net/xhmico?type=blog",  
                        "mico8080@163.com"))  
                // 版本  
                .version("1.0")  
                .build();  
    }  
  
}

import com.changgou.common.knife4jconfig.Knife4jConfiguration;  
import org.springframework.context.annotation.Import;  
  
import java.lang.annotation.*;  
  
@Target({ ElementType.TYPE })  
@Retention(RetentionPolicy.RUNTIME)  
@Documented  
@Inherited  
@Import({ Knife4jConfiguration.class })  
public @interface EnableCustomSwagger {  
}

@EnableCustomSwagger  
@SpringBootApplication  
@EnableDiscoveryClient  
@MapperScan(basePackages = {"com.changgou.goods.dao"})  
public class GoodsApplication {  
  
    public static void main(String[] args) {  
        SpringApplication.run(GoodsApplication.class,args);  
    }  
    @Bean  
    public RestTemplate restTemplate(){  
        return new RestTemplate();  
    }  
    /***  
     * IdWorker     
     * @return:  
     */  
    @Bean  
    public IdWorker idWorker(){  
        return new IdWorker(0,0);  
    }  
}
